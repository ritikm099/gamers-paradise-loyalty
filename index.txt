    const utils = {
      monthKey: ()=>{const d=new Date();return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}`},
      todayKey: ()=>{const d=new Date();return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`},
      toRad: (d)=>d*Math.PI/180,
      distance: (lat1,lon1,lat2,lon2)=>{const R=6371,dLat=utils.toRad(lat2-lat1),dLon=utils.toRad(lon2-lon1),a=Math.sin(dLat/2)**2+Math.cos(utils.toRad(lat1))*Math.cos(utils.toRad(lat2))*Math.sin(dLon/2)**2;return R*2*Math.atan2(Math.sqrt(a),Math.sqrt(1-a))},
      formatDate: (dateStr)=>{const[y,m,d]=dateStr.split('-');return new Date(y,m-1,d).toLocaleDateString('en-IN',{day:'numeric',month:'short'})},
      showAlert: (el,msg,type='error')=>{el.textContent=msg;el.className=`alert show alert-${type}`},
      hideAlert: (el)=>{el.classList.remove('show')},
      screen: (name)=>{document.querySelectorAll('.screen').forEach(s=>{s.classList.remove('active')});
      document.getElementById(name+'Screen').classList.add('active')},
      confetti: ()=>{for(let i=0;i<50;i++){const el=document.createElement('div');el.className='confetti';el.style.left=(Math.random()*100)+'%';el.style.top='0';el.style.width='8px';el.style.height='8px';el.style.background=['#ffd86f','#ff6bcb','#7b2cbf','#00e676'][Math.floor(Math.random()*4)];el.style.position='fixed';el.style.pointerEvents='none';el.style.zIndex='1000';document.body.appendChild(el);const dur=2000+Math.random()*1000;el.animate([{transform:'translateY(0)',opacity:1},{transform:`translateY(${100+Math.random()*100}vh) rotate(${Math.random()*360}deg)`,opacity:0}],{duration:dur,easing:'cubic-bezier(0.25,0.46,0.45,0.94)'});setTimeout(()=>el.remove(),dur)}}
    };

    async function stampVisit(phone){
      const uid='user_'+phone;
      const mk=utils.monthKey();
      const tk=utils.todayKey();
      const docRef=doc(db,'loyalty_users',uid);
      const snap=await getDoc(docRef);
      
      let data = snap.exists() ? snap.data() : {phone,createdAt:new Date().toISOString(),months:{}};
      if(!data.months) data.months={};
      if(!data.months[mk]) data.months[mk]={visits:0,dates:[],claimed:[],monthStartDate:new Date().toISOString()};
      
      const monthData = data.months[mk];
      if(monthData.dates.includes(tk)) return {success:false,alreadyVisited:true,visits:monthData.visits,claimed:monthData.claimed,dates:monthData.dates};
      
      monthData.visits += 1;
      monthData.dates.push(tk);
      
      const cinsRef = collection(db,'loyalty_users',uid,'checkIns');
      await addDoc(cinsRef,{date:tk,timestamp:new Date().toISOString()});
      
      const milestones = Object.keys(REWARDS).map(Number).sort((a,b)=>a-b);
      let newReward = null;
      for(const m of milestones){if(monthData.visits===m && !monthData.claimed.includes(m)){newReward={m,name:REWARDS[m]};break}}
      
      if(!snap.exists()) await setDoc(docRef,data);
      else await updateDoc(docRef,{[`months.${mk}`]:monthData});
      
      return {success:true,alreadyVisited:false,visits:monthData.visits,claimed:monthData.claimed,dates:monthData.dates,newReward};
    }

// Full visibility and validation
async function verifySessionCode(code){
  if(!code || typeof code !== 'string') {
    console.error('Invalid QR code type:', typeof code);
    return false;
  }

  let normalizedCode;
  try {
    normalizedCode = decodeURIComponent(code.trim());
  } catch (err) {
    console.warn('Failed to decode QR code, using raw value:', err.message);
    normalizedCode = code.trim();
  }

  if(!normalizedCode.toUpperCase().startsWith('SESSION_')) {
    console.error('Invalid format:', normalizedCode);
    return false;
  }

  const codeToQuery = normalizedCode.replace(/^(SESSION_)([^_]+)(_.*)$/i, (match, prefix, id, suffix) => {
    return `${prefix}${id.toLowerCase()}${suffix}`;
  });

  const sessionsRef = collection(db,'gamers_paradise_data','gamersparadise_prod','sessions');
  try {
    const q = query(sessionsRef, where('qrCode','==',codeToQuery));
    const snap = await getDocs(q);
    return !snap.empty;
  } catch(e) {
    console.error('Session query failed:', e.message);
    return false;
  }
}

    async function claimReward(phone,milestone){
      const uid='user_'+phone;
      const mk=utils.monthKey();
      const docRef=doc(db,'loyalty_users',uid);
      const snap=await getDoc(docRef);
      if(!snap.exists()) throw new Error('User not found');
      const data=snap.data();
      const monthData=data.months?.[mk];
      if(!monthData) throw new Error('Month data not found');
      if(!monthData.claimed.includes(milestone)){monthData.claimed.push(milestone);await updateDoc(docRef,{[`months.${mk}`]:monthData})}
    }

    async function fetchUserData(phone){
      const uid='user_'+phone;
      const docRef=doc(db,'loyalty_users',uid);
      const snap=await getDoc(docRef);
      if(!snap.exists()) return null;
      const data=snap.data();
      const mk=utils.monthKey();
      return {visits:data.months?.[mk]?.visits||0,claimed:data.months?.[mk]?.claimed||[],dates:data.months?.[mk]?.dates||[]};
    }

    async function fetchCustomerName(phone){
      const customersRef = collection(db,'gamers_paradise_data','gamersparadise_prod','customers');
      const q = query(customersRef, where('phone', '==', phone));
      const snap = await getDocs(q);
      if(snap.empty) return null;
      const customer = snap.docs[0].data();
      return customer.name || null;
    }

    function renderMilestones(visits,claimed,targetEl){
      targetEl.innerHTML='';
      const milestones=[3,6,9,12,15,18];
      for(const m of milestones){
        const done=visits>=m;
        const isClaimed=claimed.includes(m);
        const el=document.createElement('div');
        el.className='milestone-item';
        if(done) el.classList.add('unlocked');
        if(isClaimed) el.classList.add('claimed');
        el.innerHTML=`
          <div class="milestone-icon">${ICONS[m]}</div>
          <div class="milestone-details">
            <strong>${m} visits</strong>
            <small>${REWARDS[m]}</small>
          </div>
          <div class="milestone-status ${done && !isClaimed?'unlocked':isClaimed?'claimed':'locked'}">
            ${isClaimed?'✓ Claimed':done?'<button class="claim-btn" data-m="'+m+'">CLAIM</button>':'🔒 Locked'}
          </div>`;
        targetEl.appendChild(el);
      }
      targetEl.querySelectorAll('.claim-btn').forEach(btn=>btn.addEventListener('click',async(e)=>{

      }));
    }

    function hasCheckedInToday(dates){
      return dates.includes(utils.todayKey());
    }

    function updateTodayStatus(dates){
      const statusEl = document.getElementById('todayStatus');
      if(!statusEl) return;
      if(hasCheckedInToday(dates)){
        statusEl.textContent = 'You have already checked in today. No scan needed.';
      } else {
        statusEl.textContent = 'You have not checked in today. Scan QR to punch your visit.';
      }
    }

      async function scanFrame(){
        if(!scanActive) return;
        if(qrVideo.readyState === HTMLVideoElement.HAVE_ENOUGH_DATA){
          qrCanvas.width = qrVideo.videoWidth;
          qrCanvas.height = qrVideo.videoHeight;
          const ctx = qrCanvas.getContext('2d');
          ctx.drawImage(qrVideo,0,0,qrCanvas.width,qrCanvas.height);
          const imageData = ctx.getImageData(0,0,qrCanvas.width,qrCanvas.height);
          const code = jsQR(imageData.data, imageData.width, imageData.height, {inversionAttempts:'dontInvert'});
          if(code && code.data){
            scanStatus.textContent='QR found! Verifying...';
            scanActive=false;
            await processQRCode(code.data);
            return;
          }
        }
        requestAnimationFrame(scanFrame);
      }
      
      async function processQRCode(scannedCode){
        try{
          const valid = await verifySessionCode(scannedCode.trim());
          if(!valid){
            utils.showAlert(checkinAlert,'QR code was invalid or already used. Please try again.','error');
            stopVideoStream();
            return;
          }
          stopVideoStream();
          await completeCheckin(currentPhone);
        }catch(e){
          utils.showAlert(checkinAlert,'QR verification failed. Please try again.','error');
          stopVideoStream();
        }
      }
